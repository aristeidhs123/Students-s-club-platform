CREATE TABLE user (
	user_id INT AUTO_INCREMENT,
	name VARCHAR(50) NOT NULL,
	lastname VARCHAR(50) NOT NULL,
	email VARCHAR(50) NOT NULL,
	password VARCHAR(100) NOT NULL,
	role ENUM('user', 'manager', 'admin') DEFAULT 'user',
	profil_img VARCHAR(50) DEFAULT NULL,
	user_bio TEXT DEFAULT NULL,
	user_interests TEXT,
	PRIMARY KEY(user_id)
	
);


CREATE TABLE club (
	club_id INT AUTO_INCREMENT,
	title VARCHAR(50) NOT NULL UNIQUE,
	description TEXT NOT NULL,
	manager_id INT NOT NULL,
	status enum('pending', 'approved', 'deleted') DEFAULT 'pending',
	creationdate timestamp default current_timestamp,
	PRIMARY KEY(club_id),
	CONSTRAINT MANAGERID
	FOREIGN KEY(manager_id)
	REFERENCES user(user_id)
	ON DELETE CASCADE ON UPDATE CASCADE
);


CREATE TABLE membership(
	membership_id INT AUTO_INCREMENT,
	userid INT NOT NULL,
	clubid INT NOT NULL,
	membership_role ENUM('user', 'manager') DEFAULT 'user',
	membership_status ENUM ('pending', 'approved', 'rejected') DEFAULT 'pending',
	membershipdate timestamp default current_timestamp,
	PRIMARY KEY(membership_id),
	CONSTRAINT USERID
	FOREIGN KEY(userid)
	REFERENCES user(user_id)
	ON DELETE CASCADE ON UPDATE CASCADE,
	CONSTRAINT CLUBID
	FOREIGN KEY(clubid)
	REFERENCES club(club_id)
	ON DELETE CASCADE ON UPDATE CASCADE
	
	

);

CREATE TABLE event (
	event_id INT AUTO_INCREMENT,
	ev_clubid INT NOT NULL,
	ev_title VARCHAR(50) NOT NULL,
	ev_description TEXT NOT NULL,
	ev_role enum('public', 'private'),
	ev_seats INT,
	ev_location VARCHAR(50) NOT NULL,
	ev_gplatos DECIMAL(10,8) NOT NULL,
	ev_gmikos DECIMAL(10,8) NOT NULL,
	ev_starttime DATETIME NOT NULL,
	ev_endtime DATETIME NOT NULL,
	PRIMARY KEY(event_id),
	CONSTRAINT EVCLUBID
	FOREIGN KEY(ev_clubid)
	REFERENCES club(club_id)
	ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE reservation (
	res_id INT AUTO_INCREMENT,
	res_evid INT NOT NULL,
	res_userid INT NOT NULL,
	res_status ENUM ('pending', 'approved', 'rejected') DEFAULT 'pending',
	res_apprearedatevent BOOLEAN DEFAULT FALSE,
	reservationdate timestamp default current_timestamp,
	PRIMARY KEY(res_id ),
	CONSTRAINT RUSERID
	FOREIGN KEY(res_userid)
	REFERENCES user(user_id)
	ON DELETE CASCADE ON UPDATE CASCADE,
	CONSTRAINT RCLUBID
	FOREIGN KEY(res_evid)
	REFERENCES event(event_id)
	ON DELETE CASCADE ON UPDATE CASCADE

);

CREATE TABLE chatbot_messages (
    message_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    user_problem TEXT NOT NULL,
    bot_answer TEXT NOT NULL,
    needs_admin BOOLEAN DEFAULT FALSE,
    conversation_status ENUM('BOT_ONLY', 'WAITING_ADMIN', 'ANSWERED', 'CLOSED') DEFAULT 'BOT_ONLY',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id) REFERENCES user(user_id)
    ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE contact_admin_messages (
    contact_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    chatbot_message_id INT,
    full_name VARCHAR(100),
    email VARCHAR(100),
    problem TEXT NOT NULL,
    admin_answer TEXT,
    is_read BOOLEAN DEFAULT FALSE,
    status ENUM('PENDING', 'ANSWERED', 'CLOSED') DEFAULT 'PENDING',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    answered_at DATETIME NULL,

    FOREIGN KEY (user_id) REFERENCES user(user_id)
    ON DELETE CASCADE ON UPDATE CASCADE,

    FOREIGN KEY (chatbot_message_id) REFERENCES chatbot_messages(message_id)
    ON DELETE SET NULL ON UPDATE CASCADE
);

CREATE TABLE internal_schedule (
    schedule_id INT AUTO_INCREMENT PRIMARY KEY,
    club_id INT NOT NULL,
    created_by INT NOT NULL,
    title VARCHAR(100) NOT NULL,
    description TEXT,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    status ENUM('draft', 'active', 'completed', 'cancelled') DEFAULT 'draft',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT SCH_CLUBID
    FOREIGN KEY (club_id)
    REFERENCES club(club_id)
    ON DELETE CASCADE ON UPDATE CASCADE,

    CONSTRAINT SCH_USERID
    FOREIGN KEY (created_by)
    REFERENCES user(user_id)
    ON DELETE CASCADE ON UPDATE CASCADE
);

    ON DELETE CASCADE ON UPDATE CASCADE
